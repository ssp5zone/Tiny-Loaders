Tiny Loaders

This is a demo of various Tiny AMD Module Loaders.

                            {
                                "dependencies": {

                                    // only get almond
                                    "almond": "^0.3.3",
                                    // for styling
                                    "bootstrap": "3.3.7",
                                    // Curl.js
                                    "curl-amd": "^0.8.12",
                                    "jquery": "^3.3.1",
                                    // for Styling code
                                    "prismjs": "^1.11.0"
                                },
                                "devDependencies": {
                                    // CSS Loader for Webpack
                                    "css-loader": "^0.28.9",
                                    // Raw file Loader for Webpack
                                    "file-loader": "^1.1.7",
                                    // HTML Loader for Webpack
                                    "html-loader": "^0.5.5",
                                    // Didn't work. Bundles only 1 level of dependencies.
                                    "js-bundler": "^1.2.3",
                                    // Small server for serving file.
                                    "lite-server": "^2.3.0",
                                    // Cool but doesn't work either. Need to check
                                    "parcel-bundler": "^1.6.1",
                                    // Rollup is basically for bundling CJS files
                                    "rollup": "^0.56.2",
                                    // but it also has an AMD plugin
                                    "rollup-plugin-amd": "^3.0.0",
                                    // Style Loader for Webpack
                                    "style-loader": "^0.20.2",
                                    // This one works.
                                    "webpack": "^3.11.0"
                                },
                                "scripts": {
                                    // Build using rollup
                                    "bundle-rollup": "rollup --config rollup.bundle.js",
                                    // Build using JS Bundler
                                    "js-bundler": "bundle public/app/main-almond.js > dist/js-bundle/bundle.js",
                                    // Build using Parcel.js
                                    "parcel-build": "parcel build public/app/main-almond.js -d dist/parcel --no-minify --no-cache",
                                    // Build using Webpack
                                    // If no --config is given it looks for webpack.config.js in current path
                                    "webpack-bundle": "webpack",          
                                    // run server                          
                                    "server": "lite-server",
                                    // install, build and run server
                                    "start": "npm install && npm run webpack-bundle && npm run server"
                                }
                            }
